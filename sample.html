class Girlfriend
def initialize(hash)
  @name = hash[:name]
  @age = hash[:age]
  @height = hash[:height]
  @weight = hash[:weight]
  @gender = hash[:gender]
end
end

Inheritance

class Wife < Girlfriend
attr_accessor :husband, :bank_account
def initialize(hash)
  @husband = hash[:husband]
  @bank_account = hash[:bank_account]
  super
end
def rob_salary
  puts "#{@name}takes #{@husband}'s salary away!"
end
end
